.PHONY: help install install-system install-python setup run clean test lint format docker-build docker-run security-check

# Colors
GREEN := \033[0;32m
RED := \033[0;31m
YELLOW := \033[1;33m
NC := \033[0m

# Default target
help:
	@echo "$(GREEN)PHISH-BOT v2.0.0 - Available Commands$(NC)"
	@echo "========================================="
	@echo "$(YELLOW)make install$(NC)       - Full installation (system + Python)"
	@echo "$(YELLOW)make install-system$(NC) - Install system dependencies"
	@echo "$(YELLOW)make install-python$(NC) - Install Python dependencies"
	@echo "$(YELLOW)make setup$(NC)         - Setup virtual environment"
	@echo "$(YELLOW)make run$(NC)           - Run PHISH-BOT"
	@echo "$(YELLOW)make run-sudo$(NC)      - Run PHISH-BOT with sudo"
	@echo "$(YELLOW)make clean$(NC)         - Clean temporary files"
	@echo "$(YELLOW)make test$(NC)          - Run tests"
	@echo "$(YELLOW)make lint$(NC)          - Run linters"
	@echo "$(YELLOW)make format$(NC)        - Format code"
	@echo "$(YELLOW)make docker-build$(NC)  - Build Docker image"
	@echo "$(YELLOW)make docker-run$(NC)    - Run Docker container"
	@echo "$(YELLOW)make security-check$(NC)- Run security checks"
	@echo "========================================="

# Full installation
install: setup install-system install-python
	@echo "$(GREEN)✓ Installation complete!$(NC)"

# Install system dependencies
install-system:
	@echo "$(GREEN)Installing system dependencies...$(NC)"
	@if command -v apt-get > /dev/null; then \
		sudo apt-get update && sudo apt-get install -y nmap hping3 dsniff macchanger net-tools iproute2 iptables tcpdump traceroute dnsutils whois curl wget git nikto sqlmap gobuster chromium-browser chromium-chromedriver libpcap-dev; \
	elif command -v brew > /dev/null; then \
		brew install nmap hping3 dsniff macchanger net-tools tcpdump traceroute bind whois curl wget git nikto sqlmap gobuster chromium chromedriver libpcap; \
	else \
		echo "$(RED)No supported package manager found$(NC)"; \
	fi

# Install Python dependencies
install-python:
	@echo "$(GREEN)Installing Python dependencies...$(NC)"
	@pip install --upgrade pip setuptools wheel
	@pip install -r requirements.txt

# Setup virtual environment
setup:
	@echo "$(GREEN)Setting up virtual environment...$(NC)"
	@python3 -m venv .venv
	@. .venv/bin/activate && pip install --upgrade pip
	@echo "$(GREEN)Virtual environment created. Run 'source .venv/bin/activate' to activate$(NC)"

# Run PHISH-BOT
run:
	@echo "$(GREEN)Starting PHISH-BOT...$(NC)"
	@python3 phishbot.py

# Run with sudo
run-sudo:
	@echo "$(GREEN)Starting PHISH-BOT with elevated privileges...$(NC)"
	@sudo .venv/bin/python phishbot.py

# Clean temporary files
clean:
	@echo "$(GREEN)Cleaning temporary files...$(NC)"
	@find . -type d -name "__pycache__" -exec rm -rf {} + 2>/dev/null || true
	@find . -type f -name "*.pyc" -delete
	@find . -type f -name "*.pyo" -delete
	@find . -type f -name "*.log" -delete
	@rm -rf .pytest_cache .coverage htmlcov build dist *.egg-info
	@echo "$(GREEN)Clean complete$(NC)"

# Run tests
test:
	@echo "$(GREEN)Running tests...$(NC)"
	@pip install pytest pytest-cov
	@python -m pytest --cov=. --cov-report=term-missing

# Lint code
lint:
	@echo "$(GREEN)Running linters...$(NC)"
	@pip install flake8 black mypy
	@flake8 . --count --max-complexity=10 --statistics
	@mypy . --ignore-missing-imports

# Format code
format:
	@echo "$(GREEN)Formatting code...$(NC)"
	@black .

# Docker build
docker-build:
	@echo "$(GREEN)Building Docker image...$(NC)"
	@docker build -t phishbot:latest .

# Docker run
docker-run:
	@echo "$(GREEN)Running Docker container...$(NC)"
	@docker run -it --rm \
		--network host \
		--privileged \
		-v $(PWD)/.phishbot:/app/.phishbot \
		-v $(PWD)/logs:/app/logs \
		--name phishbot \
		phishbot:latest

# Docker compose up
docker-compose-up:
	@echo "$(GREEN)Starting Docker Compose...$(NC)"
	@docker-compose up -d

# Docker compose down
docker-compose-down:
	@echo "$(GREEN)Stopping Docker Compose...$(NC)"
	@docker-compose down

# Security check
security-check:
	@echo "$(GREEN)Running security checks...$(NC)"
	@pip install bandit safety
	@bandit -r . -f json -o bandit-report.json || true
	@safety check --json > safety-report.json || true
	@echo "$(GREEN)Security reports generated: bandit-report.json, safety-report.json$(NC)"

# Update dependencies
update:
	@echo "$(GREEN)Updating dependencies...$(NC)"
	@pip install --upgrade -r requirements.txt

# Show version
version:
	@echo "PHISH-BOT v2.0.0"

# Create .env from example
env:
	@if [ ! -f .env ]; then cp .env.example .env; echo "$(GREEN).env file created$(NC)"; fi

# Development setup
dev-setup: setup env install-python
	@echo "$(GREEN)Development environment ready!$(NC)"

# Production setup
prod-setup: install clean
	@echo "$(GREEN)Production environment ready!$(NC)"
